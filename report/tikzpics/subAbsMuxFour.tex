\def\absMuxFourWidth{3}
\def\absMuxFourHeight{6}

\subtikzpicturedef{subAbsMuxFour} {
    l0, l1, l2, l3,
    b0, b1, r0,
    north, south, east, west,
    northeast, northwest, southeast, southwest,
    center,
    origin%
} {
    \draw (#1-start) coordinate (#1-origin);
    \subAbsCtkDiagramBase {#1-absCtkDiagramBase} {#1-origin} {origin}

    \draw [absCtkModule]
    (#1-origin) coordinate (#1-bottomLeftCorner)
    (#1-bottomLeftCorner) ++(\absMuxFourWidth, \absMuxFourHeight)
    coordinate (#1-topRightCorner)

    \markgeocoordinate {#1}
    {(#1-topRightCorner)} {(#1-bottomLeftCorner)}
    {(#1-bottomLeftCorner)} {(#1-topRightCorner)}

    (#1-northwest) ++(-\absCtkPinPad, 0)
    ++(0, -0.5) coordinate (#1-l0) ++(0, -0.25)
    ++(0, -0.5) coordinate (#1-l1) ++(0, -0.25)
    ++(0, -0.5) coordinate (#1-l2) ++(0, -0.25)
    ++(0, -0.5) coordinate (#1-l3)

    (#1-southwest) ++(0.5, -\absCtkPinPad)
    coordinate (#1-b0)
    ++(0.5, 0) coordinate (#1-b1)

    (#1-northeast) ++(\absCtkPinPad, -0.5)
    coordinate (#1-r0)

    (#1-bottomLeftCorner) rectangle (#1-topRightCorner)

    (#1-center) coordinate (#1-muxCenter)

    \markgeocoordinate {#1}
    {(#1-north)} {(#1-b0)}
    {(#1-l0)} {(#1-r0)}

    ;

}
