\definecolor{seedColorPrimary} {RGB} {161, 114, 96}
% Hazelnut
% #A17260

\definecolor{seedColorSecondary} {RGB} {179, 137, 113}
% Macaroon
% #B38971


\subtikzpicturedef{subSeedColor} {
    north, south, east, west,
    northeast, northwest, southeast, southwest,
    center,
    origin%
} {
    \draw (#1-start) coordinate (#1-origin);

    \subSeedSlice {#1-seedSlice} {#1-origin} {origin}

    \fill [fill = seedColorPrimary, rounded corners = 1pt]
    \ovrDrawSeedLeftSlice {#1-seedSlice}
    ;

    \fill [fill = seedColorSecondary, rounded corners = 1pt]
    \ovrDrawSeedRightSlice {#1-seedSlice}
    ;

    \draw
    \markgeocoordinate {#1}
    {(#1-seedSlice-north)} {(#1-seedSlice-south)}
    {(#1-seedSlice-west)} {(#1-seedSlice-east)}
    ;

}

\subtikzpictureactivate{subSeedColor}
