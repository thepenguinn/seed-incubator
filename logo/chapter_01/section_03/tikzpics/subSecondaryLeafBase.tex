\subtikzpicturedef{subSecondaryLeafBase} {
    north, south, east, west,
    northeast, northwest, southeast, southwest,
    center,
    outerLeftBottomCorner,
    outerLeftMostCorner,
    outerLeftTopCorner,
    outerRightTopCorner,
    outerRightMostCorner,
    outerRightBottomCorner,
    origin%
} {
    \draw (#1-start) coordinate (#1-origin);
    \subLeafBase {#1-subLeafBase} {#1-origin} {origin}

    \draw

    %% left side

    (#1-origin)
    coordinate (#1-outerLeftBottomCorner)
    ++(\leafBaseAngle:\leafBaseSecondaryEdgeLength)
    coordinate (#1-outerLeftMostCorner)

    (#1-outerLeftMostCorner)
    ++({180 - \leafBaseAngle}:\leafBaseSecondaryEdgeLength)
    coordinate (#1-outerLeftTopCorner)

    %% right side

    (#1-outerLeftBottomCorner)
    ++(\leafBaseThickness, 0)
    coordinate (#1-outerRightBottomCorner)

    (#1-outerRightBottomCorner)
    ++({180 - \leafBaseAngle}:\leafBaseSecondaryEdgeLength)
    coordinate (#1-outerRightMostCorner)

    (#1-outerRightMostCorner)
    ++(\leafBaseAngle:\leafBaseSecondaryEdgeLength)
    coordinate (#1-outerRightTopCorner)

    ;

    \draw
    \markgeocoordinate {#1}
    {(#1-outerRightTopCorner)} {(#1-outerRightBottomCorner)}
    {(#1-outerLeftMostCorner)} {(#1-outerRightMostCorner)}
    ;

}

\subtikzpictureactivate{subSecondaryLeafBase}

\newcommand\ovrConnectSecondaryLeafBase[1] {
    (#1-outerLeftBottomCorner)
    -- (#1-outerLeftMostCorner)
    -- (#1-outerLeftTopCorner)
    -- (#1-outerRightTopCorner)
    -- (#1-outerRightMostCorner)
    -- (#1-outerRightBottomCorner)
    -- cycle
}
