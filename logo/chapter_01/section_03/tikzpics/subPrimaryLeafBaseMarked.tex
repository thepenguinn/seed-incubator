\subtikzpicturedef{subPrimaryLeafBaseMarked} {
    north, south, east, west,
    northeast, northwest, southeast, southwest,
    center,
    origin%
} {
    \draw (#1-start) coordinate (#1-origin);

    \subPrimaryLeafBase {#1-primaryLeafBase} {#1-origin} {origin}

    \fill [leafBaseStyle]
    \ovrConnectPrimaryLeafBase {#1-primaryLeafBase}
    ;

    %% length, angle

    \draw [leafOuterMarkStyle]
    (#1-primaryLeafBase-outerLeftTopCorner)
    ++({270 - \leafBaseAngle}:\leafMarkPad)
    coordinate (#1-lineMarkFirst)

    (#1-primaryLeafBase-outerLeftTopCorner)
    ++({270 - \leafBaseAngle}:1)
    coordinate (#1-lineMarkSecond)

    (#1-primaryLeafBase-outerLeftMostCorner)
    ++({270 - \leafBaseAngle}:\leafMarkPad)
    coordinate (#1-lineMarkFourth)

    (#1-primaryLeafBase-outerLeftMostCorner)
    ++({270 - \leafBaseAngle}:1)
    coordinate (#1-lineMarkThird)

    (#1-lineMarkFirst)
    node [ocirc] {}
    -- (#1-lineMarkSecond)
    -- (#1-lineMarkThird)
    -- (#1-lineMarkFourth)
    node [ocirc] {}

    ;

    \path [leafOuterMarkStyle]
    (#1-lineMarkSecond)
    -- (#1-lineMarkThird)
    node [midway, above = 4pt, sloped] {\leafBaseEdgeLength}
    ;

    %% angle

    \draw [leafInnerMarkStyle]
    (#1-primaryLeafBase-outerRightBottomCorner)
    arc (0:\leafBaseAngle:\leafBaseThickness)

    (#1-primaryLeafBase-outerLeftBottomCorner)
    ++({\leafBaseAngle / 2}:\leafBaseThickness)
    -- ++(0, 2)
    node [ocirc] {}
    node [above = 4pt] {\leafBaseAngle$^\circ$}
    ;

    %% corner radius

    \draw [leafInnerMarkStyle]
    (#1-primaryLeafBase-outerLeftMostCorner)
    ++({2 * \leafBaseCornerRadiusInCm}, 0)
    coordinate (#1-cornerCenter)

    (#1-cornerCenter)
    circle ({2 * \leafBaseCornerRadiusInCm})
    (#1-cornerCenter)
    -- ++(2, 0)
    node [ocirc] {}
    node [right = 4pt] {\leafBaseCornerRadius}
    ;

    \draw
    \markgeocoordinate {#1}
    {(#1-primaryLeafBase-north)} {(#1-primaryLeafBase-south)}
    {(#1-primaryLeafBase-west)} {(#1-primaryLeafBase-east)}
    ;

}

\subtikzpictureactivate{subPrimaryLeafBaseMarked}
